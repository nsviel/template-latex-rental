\newcommand{\docpages}[4]{
    \newpage

    \addwatermark
    \foreach \p in {#3,...,#4} {
        \ifnum\p=#3
            \subsection{\textcolor{black!50}{#1} - Page \p}
        \else
            \newpage
            \addwatermark
            \subsection{\textcolor{black!50}{#1} - Page \p}
        \fi
        \begin{figure}[H]
            \centering
            \includegraphics[page=\p, width=0.9\linewidth]{#2}
        \end{figure}
    }
}
